	[EmailID] As [Order],
	[EmailID] As [EmailID],

Union All

--I have decided to keep a 'Single Source Of Truth' for the Conversation_IDs on the Email table, and not also on the Email Reply table, in order to avoid possible issues where one table has Conversation_IDs updated, but the other doesn't, and so we would experience a data pollution error, where the table data when taken together, conflicts...

--The results you need...
--USER-ASSIGNED-GLOBAL-VARIABLES
--USER-ASSIGNED-GLOBAL-VARIABLES
--USER-ASSIGNED-GLOBAL-VARIABLES
Declare @Email_Table_Seed As Bigint
Set @Email_Table_Seed = 1 --A value of one (1) refers to [EmailID] table, whereas a value of two (2) refers to [EmailReplyID] table...
Declare @Email_ID_Seed As Bigint
Set @Email_ID_Seed = 3
--USER-ASSIGNED-GLOBAL-VARIABLES
--USER-ASSIGNED-GLOBAL-VARIABLES
--USER-ASSIGNED-GLOBAL-VARIABLES

Declare @Conversation_ID_Seed As Bigint
If @Email_Table_Seed = 1
	Begin
		Set @Conversation_ID_Seed = (Select [Conversation_ID] From @tblEmail_Asymmetric_Interleaving_Chain_Discovery Where [Email_ID] = @Email_ID_Seed)
	End
Else If @Email_Table_Seed = 2
	Begin
		Set @Conversation_ID_Seed = (Select B.[Conversation_ID] From @tblReply_Asymmetric_Interleaving_Chain_Discovery A Inner Join @tblEmail_Asymmetric_Interleaving_Chain_Discovery B On (A.[Email_Reply_ID] = @Email_ID_Seed And B.[Email_ID] = A.[Email_ID]))
	End

Declare @Min_In_Conversation_Email_Table_Email_ID As Bigint
Set @Min_In_Conversation_Email_Table_Email_ID = (Select Min([Email_ID]) From @tblEmail_Asymmetric_Interleaving_Chain_Discovery Where [Conversation_ID] = @Conversation_ID_Seed)

Declare @Intermediate_Results_Asymmetric_Interleaving_Chain_Discovery Table ([Conversation_ID] Bigint, [Email_ID] Bigint, [Email_Reply_ID] Bigint, [Subject] Varchar(Max))
Insert Into @Intermediate_Results_Asymmetric_Interleaving_Chain_Discovery ([Conversation_ID], [Email_ID], [Email_Reply_ID], [Subject])
Select
	A.[Conversation_ID] As [Conversation_ID],
	A.[Email_ID] As [Emai_lID],
	Null As [Email_Reply_ID],
	A.[Subject] As [Subject]
From @tblEmail_Asymmetric_Interleaving_Chain_Discovery A
Union All
Select
	C.[Conversation_ID] As [Conversation_ID],
	B.[Email_ID] As [Email_ID],
	B.[Email_Reply_ID] As [Email_Reply_ID],
	B.[Subject] As [Subject]
From @tblReply_Asymmetric_Interleaving_Chain_Discovery B
Inner Join @tblEmail_Asymmetric_Interleaving_Chain_Discovery C
	On (C.[Conversation_ID] = @Conversation_ID_Seed And C.[Email_ID] = B.[Email_ID])

Select
	(Case When A.[Email_Reply_ID] Is Not Null Then Null Else A.[Email_ID] End) As [Email_ID],
	A.[Email_Reply_ID] As [Email_Reply_ID],
	A.[Subject] As [Subject]
From @Intermediate_Results_Asymmetric_Interleaving_Chain_Discovery A
Where A.[Conversation_ID] = @Conversation_ID_Seed And A.[Email_ID] <> @Min_In_Conversation_Email_Table_Email_ID Or A.[Email_Reply_ID] Is Not Null
Order By A.[Email_ID] Asc, A.[Email_Reply_ID] Asc
