But be aware, that it is an absolute 'Compute Hog'!!!

Declare @Intermediate_Results Table ([Order] Bigint, [EmailID] Bigint, [EmailReplyID] Bigint, [Subject] Varchar(Max))
Insert Into @Intermediate_Results ([Order], [EmailID], [EmailReplyID], [Subject])
Select
	[EmailID] As [Order],
	[EmailID] As [EmailID],
	Null As [EmailReplyID],
	[Subject] As [Subject]
From @tblEmail
Where [EmailID] <> 1
Union All
Select
	[EmailReplyID] As [Order],
	Null As [EmailID],
	[EmailReplyID] As [EmailReplyID],
	[Subject] As [Subject]
From @tblReply

Select
	[EmailID] As [EmailID],
	[EmailReplyID] As [EmailReplyID],
	[Subject] As [Subject]
From @Intermediate_Results
Order By [Order] Asc, [EmailReplyID] Asc, [EmailID] Asc

Declare @Conversation_ID_Seed As Bigint
Set @Conversation_ID_Seed = 1

Declare @Email_Table_Seed As Bigint
Set @Email_Table_Seed = 2 --A value of one (1) refers to [EmailID] table, whereas a value of two (2) refers to [EmailReplyID] table...

Declare @Email_ID_Seed As Bigint
Set @Email_ID_Seed = 3

Declare @Email_Chain_Seed Table ([Conversation_ID_Seed] Bigint, [Email_Table_Seed] Bigint, [Email_ID_Seed] Bigint)
Insert Into @Email_Chain_Seed ([Conversation_ID_Seed], [Email_Table_Seed], [Email_ID_Seed])
Values
(@Conversation_ID_Seed, @Email_Table_Seed, @Email_ID_Seed)

--This is a powerful 'Blatt_Table', which allows us to avoid a slow loop...
Declare @Email_Chain_Blatt_Table Table ([Conversation_ID_Seed] Bigint, [Email_Table_Seed] Bigint, [Email_ID_Seed] Bigint, [Conversation_ID_B] Bigint, [Email_ID_B] Bigint, [Email_Reply_ID_B] Bigint, [Subject_B] Varchar(Max), [Conversation_ID_C] Bigint, [Email_ID_C] Bigint, [Email_Reply_ID_C] Bigint, [Subject_C] Varchar(Max))
Insert Into @Email_Chain_Blatt_Table ([Conversation_ID_Seed], [Email_Table_Seed], [Email_ID_Seed], [Conversation_ID_B], [Email_ID_B], [Email_Reply_ID_B], [Subject_B], [Conversation_ID_C], [Email_ID_C], [Email_Reply_ID_C], [Subject_C])
Select
	[Conversation_ID_Seed] As [Conversation_ID_Seed],
	[Email_Table_Seed] As [Email_Table_Seed],
	[Email_ID_Seed] As [Email_ID_Seed],
	B.[Conversation_ID] As [Conversation_ID_B],
	B.[Email_ID] As [Email_ID_B],
	B.[Email_Reply_ID] As [Email_Reply_ID_B],
	B.[Subject] As [Subject_B],
	C.[Conversation_ID] As [Conversation_ID_C],
	C.[Email_ID] As [Email_ID_C],
	C.[Email_Reply_ID] As [Email_Reply_ID_C],
	C.[Subject] As [Subject_C]
From @Email_Chain_Seed A
Cross Join @Intermediate_Results_Asymmetric_Interleaving_Chain_Discovery B
Cross Join @Intermediate_Results_Asymmetric_Interleaving_Chain_Discovery C
Where C.[Email_Reply_ID] Is Null And C.[Conversation_ID] = A.[Conversation_ID_Seed] And B.[Conversation_ID] = A.[Conversation_ID_Seed] And C.[Email_ID] = (B.[Email_ID] - 1) Or C.[Email_Reply_ID] Is Null And C.[Conversation_ID] = A.[Conversation_ID_Seed] And B.[Conversation_ID] = A.[Conversation_ID_Seed] And C.[Email_ID] = (B.[Email_ID] + 1) Or C.[Email_Reply_ID] Is Not Null And B.[Conversation_ID] = A.[Conversation_ID_Seed] And C.[Email_ID] = (B.[Email_ID] - 1) Or C.[Email_Reply_ID] Is Not Null And B.[Conversation_ID] = A.[Conversation_ID_Seed] And C.[Email_ID] = (B.[Email_ID] + 1)                 

Declare @Max_Blatt_Table_Email_ID As Bigint
Set @Max_Blatt_Table_Email_ID = (Select Max([Email_ID_B]) From @Email_Chain_Blatt_Table)

Declare @Email_Chain_Blatt_Table_Cleansed Table ([Email_ID] Bigint, [Email_Reply_ID] Bigint, [Subject] Varchar(Max))
Insert Into @Email_Chain_Blatt_Table_Cleansed ([Email_ID], [Email_Reply_ID], [Subject])
Select
	[Email_ID_C] As [Email_ID],
	[Email_Reply_ID_C] As [Email_Reply_ID],
	Min([Subject_C]) As [Subject]
From @Email_Chain_Blatt_Table
Where [Email_ID_C] <= @Max_Blatt_Table_Email_ID
Group By [Email_ID_C], [Email_Reply_ID_C]
--I chose to apply a Min() aggregation function to the subject column, rather than simply grouping by another dimension (obviously the subject column), because it makes it possible to avoid multiple group combination checks per input row, and to simply check for the always-the-case single value in each double grouping combination (without therefore a need to sort, which would likely be slower than simply apply another grouping dimension instead of the Min() aggregation function)...

Select
	(Case When A.[Email_Reply_ID] Is Not Null Then Null Else A.[Email_ID] End) As [Email_ID],
	A.[Email_Reply_ID] As [Email_Reply_ID],
	A.[Subject] As [Subject]
From @Email_Chain_Blatt_Table_Cleansed A
Order By A.[Email_ID] Asc, A.[Email_Reply_ID] Asc
